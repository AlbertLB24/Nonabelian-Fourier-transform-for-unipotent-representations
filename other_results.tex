\section{Other interesting results}

\begin{lemma}
    Let $G$ be a group and let $H$ be a subgroup. Suppose that $G$ acts on another group $N$ by $G\rightarrow\Aut(N)$. Then the following square is commutative.

     \[
        \begin{tikzcd}
        \Rep(H\ltimes N) \arrow[r, "\Ind_{H\ltimes N}^{G\ltimes N}"] & \Rep(G\ltimes N) \\
        \Rep(H) \arrow[r, "\Ind_H^G"] \arrow[u, "\mathrm{Inf}_H^{H\ltimes N}"] & \Rep(G) \arrow[u, "\mathrm{Inf}_G^{G\ltimes N}"]
        \end{tikzcd}
    \]
\end{lemma}


\begin{cor}\label{cor:ind_inf}
    Let $W(B_n)=\langle w_{\alpha_1},\ldots,w_{\alpha_n}\rangle$ be the Weyl group of type $B_n$. There is a unique quotient map $\phi:W(B_n)\twoheadrightarrow W(A_{n-1})$ containing all short reflections in the $\ker\phi$ with a natural section that identifies $W(A_{n-1})\cong\langle w_{\alpha_2},\ldots,w_{\alpha_n}\rangle$. Finally, let $W'=\langle w_{\alpha_1},\ldots,w_{\alpha_{n-1}},w_{\alpha_0}\rangle\cong W(B_{n-1})\times C_2$. Then the following diagram commutes. 

    \[
        \begin{tikzcd}
        \Rep(W') \arrow[r, "\Ind"] & \Rep(W(B_{n})) \\
        \Rep(W(A_{n-2})) \arrow[r, "\Ind"] \arrow[u, "\mathrm{Inf}"] & \Rep(W(A_{n-1})) \arrow[u, "\mathrm{Inf}"]
        \end{tikzcd}
    \]
\end{cor}
\begin{proof}
    Apply the previous Lemma with $G=W(A_{n-1})=\langle w_{\alpha_2},\ldots,w_{\alpha_n}\rangle\cong S_n$, $H=\langle w_{\alpha_2},\ldots,w_{\alpha_{n-1}}\rangle\cong S_{n-1}$ and $N=\langle\textrm{short reflections of }W(B_n)\rangle=\langle\langle w_{\alpha_1}\rangle\rangle\cong C_2^n$ together with the natural action $S_n\rightarrow\Aut(C_2^n)$ by permutations on each coordinate.
\end{proof}

The above Corollary is useful to compute inductions if we have a nice description of the inflation maps.

\begin{proposition}\label{prop:Winf}
    Let $n\geq 2$. Then the inflation map $\mathrm{Inf}:\Irr(W(A_{n-1}))\rightarrow \Irr(W(B_n))$ send the irreducible $W(A_{n-1})$-representation labelled by $(\alpha)$ to the irreducible $W(B_n)$-representation labelled by $(\alpha,\emptyset)$, where $\alpha$ is a partition of $[n]$.
\end{proposition}
\begin{proof}
    Fix a partition $\alpha=(\alpha_1,\alpha_2,\ldots)$ of $[n]$ and let $\alpha^*=(\alpha_1^*,\alpha_2^*,\ldots)$ be its dual partition. Let $\Phi'$ be a subroot system of $\Phi$ of type $D_{\alpha_1^*}+D_{\alpha_2^*}+\cdots$. Then we have that $\Phi'$ is simply laced, and only consists of long roots so its Weyl group $W'$ is generated by long reflections too.

    \textcolor{red}{Idea of the remaining of the proof} By taking integral basis of each $D_{\alpha_i^*}$, we can take an appropriate basis $\{x_1,\ldots,x_n\}$ of $V^*$ and disjoint subsets $J_1,\ldots,J_r\subseteq[n]$ such that the sign representation of $W'$ is given by the one dimensional subspace spanned by 
    $$\prod_{k=1}^r\prod_{i,j\in J_k,\ i<j}(x_i^2-x_j^2).$$
    Then it should be able to relate the generated representation to the one obtained from $(\alpha)$ as a $W(A_{n-1})$.
\end{proof}

With these results, we can now reprove Lemma \ref{lem:indWs1}. By Corollary \ref{cor:ind_inf}, 
$$\Ind_{W_{s_1}}^{W(B_n)}\mathbf{1}=Inf_{W(A_{n-1})}^{W(B_n)}\Ind_{W(A_{n-2})}^{W(A_{n-1})}\mathbf{1}$$
A simple computation shows that 
$\Ind_{W(A_{n-2})}^{W(A_{n-1})}\mathbf{1}=\{f:S_{n-1}\backslash S_{n}\rightarrow\CC\}$, which decomposes as a direct sum of the $S_n$-invariant subspaces 
$$U_1=\{f:S_{n-1}\backslash S_{n}\rightarrow\CC\ |\ f \text{ constant}\},\quad U_2=\{f:S_{n-1}\backslash S_{n}\rightarrow\CC\ |\ \sum_{\sigma\in S_{n-1}\backslash S_{n}}f(\sigma)=0\},$$
both of which are irreducible and correspond to the $S_n$-representations labelled by $(n)$ and $((n-1)1)$, respectively. combining this with Proposition \ref{prop:Winf}, we get that
$$\Ind_{W_{s_1}}^{W(B_n)}\mathbf{1}=(n,\emptyset)+((n-1)1,\emptyset)$$
